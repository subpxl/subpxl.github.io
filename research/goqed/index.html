// ── thread-safe post store ────────────────────────────────────────────────────

var (
	mu    sync.RWMutex
	posts []string
)

func CheckLANAccess() {
	fmt.Println("== LAN access check ==")

	// 1. Check if port 8080 is listening
	conn, err := net.DialTimeout("tcp", "127.0.0.1:8080", time.Second)
	if err != nil {
		fmt.Println("FAIL: server not listening on :8080")
		return
	}
	conn.Close()
	fmt.Println("OK  : server listening on :8080")

	// 2. Check Windows network profile (PowerShell)
	out, err := exec.Command(
		"powershell",
		"-Command",
		"(Get-NetConnectionProfile).NetworkCategory",
	).Output()
	if err != nil || !strings.Contains(string(out), "Private") {
		fmt.Println("FAIL: network profile is not Private")
		return
	}
	fmt.Println("OK  : network profile is Private")

	// 3. Check firewall rule exists and is enabled
	out, err = exec.Command(
		"powershell",
		"-Command",
		"Get-NetFirewallRule -DisplayName 'Local Blog :8080' | Select -Expand Enabled",
	).Output()
	if err != nil || !strings.Contains(string(out), "True") {
		fmt.Println("FAIL: firewall rule missing or disabled")
		return
	}
	fmt.Println("OK  : firewall rule enabled")

	fmt.Println("PASS: LAN access should work")
}

// ── helpers ───────────────────────────────────────────────────────────────────

func getLocalIP() string {
	// Dial an external address (doesn't actually send packets) to find
	// which local interface the OS would use — most reliable method.
	conn, err := net.Dial("udp", "8.8.8.8:80")
	if err != nil {
		// fallback: walk interfaces
		addrs, _ := net.InterfaceAddrs()
		for _, addr := range addrs {
			if ipnet, ok := addr.(*net.IPNet); ok && !ipnet.IP.IsLoopback() {
				if ipnet.IP.To4() != nil {
					return ipnet.IP.String()
				}
			}
		}
		return "127.0.0.1"
	}
	defer conn.Close()
	return conn.LocalAddr().(*net.UDPAddr).IP.String()
}

// ── main ──────────────────────────────────────────────────────────────────────

func main() {
	localIP := getLocalIP()
	port := 8080
	url := fmt.Sprintf("http://%s:%d", localIP, port)
	CheckLANAccess()
	// Generate QR code bytes ONCE at startup — fail loudly if it errors
	qrBytes, err := qrcode.Encode(url, qrcode.Medium, 256)
	if err != nil {
		log.Fatalf("QR encode failed: %v", err)
	}

	// ── HTTP handlers ─────────────────────────────────────────────────────────

	http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
		if r.Method == http.MethodPost {
			if content := r.FormValue("content"); content != "" {
				mu.Lock()
				posts = append(posts, content)
				mu.Unlock()
				// PRG pattern: redirect after POST so refresh doesn't re-submit
				http.Redirect(w, r, "/", http.StatusSeeOther)
				return
			}
		}
		mu.RLock()
		snapshot := make([]string, len(posts))
		copy(snapshot, posts)
		mu.RUnlock()

		// reverse so newest first
		for i, j := 0, len(snapshot)-1; i < j; i, j = i+1, j-1 {
			snapshot[i], snapshot[j] = snapshot[j], snapshot[i]
		}

		w.Header().Set("Content-Type", "text/html; charset=utf-8")
		tmpl.Execute(w, struct {
			URL   string
			Posts []string
		}{url, snapshot})
	})

	http.HandleFunc("/qr", func(w http.ResponseWriter, _ *http.Request) {
		w.Header().Set("Content-Type", "image/png")
		w.Write(qrBytes)
	})

	// ── mDNS registration — keep alive until shutdown ─────────────────────────
	// FIX: store the server reference outside the goroutine so we can shut it
	//      down explicitly, and block inside the goroutine so it stays alive.
	mdnsServer, err := zeroconf.Register(
		"LocalBlog",  // service instance name  → shows as "LocalBlog" in Bonjour/Avahi
		"_http._tcp", // service type
		"local.",     // domain
		port,
		[]string{"path=/"}, // TXT record (optional metadata)
		nil,                // use all network interfaces
	)
	if err != nil {
		log.Printf("mDNS registration failed (not fatal): %v", err)
		// not fatal — IP-based access still works
	} else {
		log.Println("mDNS registered: LocalBlog._http._tcp.local.")
	}

	// ── Start HTTP server in background ──────────────────────────────────────
	go func() {
		log.Printf("Listening on :%d", port)
		if err := http.ListenAndServe(fmt.Sprintf(":%d", port), nil); err != nil {
			log.Fatalf("HTTP server error: %v", err)
		}
	}()
	// go func() {
	// 	log.Printf("HTTPS listening on :8080")
	// 	err := http.ListenAndServeTLS(
	// 		":8080",
	// 		// "cert.pem",
	// 		// "key.pem",
	// 		"LocalBlog.local+1.pem",
	// 		"LocalBlog.local+1-key.pem",
	// 		nil,
	// 	)
	// 	if err != nil {
	// 		log.Fatal(err)
	// 	}
	// }()

	log.Printf("\n\n  📡  Local Blog is live!\n")
	log.Printf("  %-18s %s\n", "Open on this PC:", url)
	log.Printf("  %-18s %s/qr\n", "QR image:", url)
	log.Printf("  %-18s LocalBlog.local:%d  (Apple/Linux only)\n\n", "mDNS:", port)

	// ── Block until Ctrl-C / SIGTERM ──────────────────────────────────────────
	sig := make(chan os.Signal, 1)
	signal.Notify(sig, syscall.SIGINT, syscall.SIGTERM)
	<-sig

	log.Println("Shutting down…")
	if mdnsServer != nil {
		mdnsServer.Shutdown() // clean mDNS goodbye packet
	}
}

package main

import (
	"crypto/rand"
	"crypto/rsa"
	"crypto/x509"
	"crypto/x509/pkix"
	"encoding/pem"
	"math/big"
	"os"
	"time"
)

func main() {
	// Generate private key
	privateKey, err := rsa.GenerateKey(rand.Reader, 2048)
	if err != nil {
		panic(err)
	}

	// Certificate template
	serialNumber, err := rand.Int(rand.Reader, big.NewInt(1<<62))
	if err != nil {
		panic(err)
	}

	certTemplate := x509.Certificate{
		SerialNumber: serialNumber,
		Subject: pkix.Name{
			Organization: []string{"Example Org"},
			CommonName:   "localhost",
		},
		NotBefore: time.Now(),
		NotAfter:  time.Now().Add(365 * 24 * time.Hour),

		KeyUsage:              x509.KeyUsageKeyEncipherment | x509.KeyUsageDigitalSignature,
		ExtKeyUsage:           []x509.ExtKeyUsage{x509.ExtKeyUsageServerAuth},
		BasicConstraintsValid: true,

		DNSNames: []string{"localhost"},
	}

	// Create certificate
	certDER, err := x509.CreateCertificate(
		rand.Reader,
		&certTemplate,
		&certTemplate,
		&privateKey.PublicKey,
		privateKey,
	)
	if err != nil {
		panic(err)
	}

	// Write certificate to file
	certOut, err := os.Create("cert.pem")
	if err != nil {
		panic(err)
	}
	pem.Encode(certOut, &pem.Block{Type: "CERTIFICATE", Bytes: certDER})
	certOut.Close()

	// Write private key to file
	keyOut, err := os.Create("key.pem")
	if err != nil {
		panic(err)
	}
	pem.Encode(keyOut, &pem.Block{
		Type:  "RSA PRIVATE KEY",
		Bytes: x509.MarshalPKCS1PrivateKey(privateKey),
	})
	keyOut.Close()
}
