func Middleware(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		log.Println(" entering middleware")
		next.ServeHTTP(w, r)
		log.Println(" exitign middleware")

func CheckRole(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		if r.Header.Get("role") == "admin" {

			next.ServeHTTP(w, r)
			log.Println(" exitign middleware")
		} else {
			http.Error(w, "unauthorized", http.StatusUnauthorized)
		}
	})
}

package main

import (
	"fmt"
	"net/http"

	"github.com/gorilla/websocket"
	ext "github.com/reugn/go-streams/extension"
	"github.com/reugn/go-streams/flow"
)

var upgrader = websocket.Upgrader{}

type Job struct {
	Data string
	Resp chan string
}

var jobs = make(chan Job, 100)

func main() {

	fmt.Println("hre")

	for i := 0; i < 5; i++ {
		go worker(i)
	}
	http.HandleFunc("/ws", wsHandler)

	http.HandleFunc("/", reverserHandler)

	fmt.Println("rinnung on 8000")
	http.ListenAndServe(":8000", nil)
}

func wsHandler(w http.ResponseWriter, r *http.Request) {
	conn, err := upgrader.Upgrade(w, r, nil)
	if err != nil {
		return
	}
	defer conn.Close()

	for {
		_, msg, err := conn.ReadMessage()
		if err != nil {
			break
		}

		resp := make(chan string, 20)

		jobs <- Job{
			Data: string(msg),
			Resp: resp,
		}

		result := <-resp
		fmt.Println("received:", string(msg))
		conn.WriteMessage(websocket.TextMessage, []byte(result))
	}
}

func reverserHandler(w http.ResponseWriter, r *http.Request) {
	file := r.FormValue("file")

	job := Job{
		Data: file,
		Resp: nil,
	}
	select {
	case jobs <- job:
		fmt.Fprintln(w, "job queued")
	default:
		http.Error(w, "queue full", http.StatusTooManyRequests)
	}
}

func worker(id int) {
	for job := range jobs {
		fmt.Println("worker", id, "processing", job.Data)

		result := processFile(job.Data, id)

		if job.Resp != nil {
			job.Resp <- result
		}
	}
}

func processFile(file string, id int) string {
	source := ext.NewFileSource(file)
	reverseMapFlow := flow.NewMap(reverseString, 1)
	newLineMapFlow := flow.NewMap(addNewLine, 1)

	out := fmt.Sprintf("out%d.txt", id)
	sink := ext.NewFileSink(out)

	source.Via(reverseMapFlow).Via(newLineMapFlow).To(sink)

	return out
}

func reverseString(s string) string {
	runes := []rune(s)
	for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
		runes[i], runes[j] = runes[j], runes[i]
	}
	return string(runes)

}
func addNewLine(s string) string {
	return s + "\n"
}

package main

import "fmt"

func main() {

	fmt.Println(twosum([]int{2, 11, 7, 15}, 9))
}

func twosum(nums []int, target int) []int {
	m := make(map[int]int)

	for i, v := range nums {
		if idx, ok := m[target-v]; ok {
			return []int{idx, i}
		}
		m[v] = i
	}
	return nil
}

func reverseint(a int) int {
	reversed := 0
	for a != 0 {
		r := a % 10
		reversed = reversed*10 + r
		a = a / 10
	}
	fmt.Println(reversed)
	return reversed
}

func valpar(inp string) bool {
	stack := []rune{}

	for _, c := range inp {

		switch c {
		case '[', '(', '{':
			stack = append(stack, c)

		case ']', ')', '}':
			if len(stack) == 0 {
				return false
			}

			top := stack[len(stack)-1]
			if (c == ')' && top != '(') ||
				(c == ']' && top != '[') ||
				(c == '}' && top != '{') {
				return false
			}
		}

	}
	return len(stack) == 0

}

import "fmt"

func merge(a, b <-chan int) <-chan int {
	out := make(chan int)
	go func() {
		defer close(out)
		for a != nil || b != nil {
			select {
			case v, ok := <-a:
				if !ok {
					a = nil
					continue
				}
				out <- v
			case v, ok := <-b:
				if !ok {
					b = nil
					continue
				}
				out <- v
			}

		}
	}()
	return out
}

	"github.com/go-chi/chi/v5"
	"go.opentelemetry.io/contrib/instrumentation/net/http/otelhttp"
	"go.opentelemetry.io/otel"
)

func main() {
	logger.Init()

	// conn := "postgres://user:pass@localhost:5432/db"
	cfg, err := config.LoadFromVault()
	if err != nil {
		logger.Log.Error("vault load failed", slog.Any("error", err))
		os.Exit(1)
	}

	conn := cfg.ConnString()

	database, err := db.Init(conn)
	if err != nil {
		logger.Log.Error("db connection failed", slog.Any("error", err))
		os.Exit(1)
	}
	if err := db.Migrate(database); err != nil {
		logger.Log.Error("migration failed", slog.Any("error", err))

		os.Exit(1)
	}

	repo := repository.NewUserRepo(database)
	producer := kafka.NewProducer("localhost:9092", "user-events")
	svc := service.NewUserService(repo, producer)

	var tr = otel.Tracer("user-service")

	h := handler.NewUserHandler(svc, tr)

	r := chi.NewRouter()
	shutdown := otelsetup.InitTracer()
	defer shutdown()
	wrapped := otelhttp.NewHandler(r, "http-server")

	h.Routes(r)

	logger.Log.Info("server starting", slog.String("port", "8080"))
	if err := http.ListenAndServe(":8080", wrapped); err != nil {
		logger.Log.Error("server failed", slog.Any("error", err))
	}

}

package config

import (
	"context"
	"fmt"

	vault "github.com/hashicorp/vault/api"
)

type DBConfig struct {
	User     string
	Password string
	Host     string
	Port     string
	Name     string
}

func LoadFromVault() (*DBConfig, error) {
	cfg := vault.DefaultConfig()
	cfg.Address = "http://localhost:8200"

	client, err := vault.NewClient(cfg)
	if err != nil {
		return nil, err
	}

	client.SetToken("root")

	secret, err := client.KVv2("secret").Get(context.Background(), "db")
	if err != nil {
		return nil, err
	}

	data := secret.Data

	return &DBConfig{
		User:     data["user"].(string),
		Password: data["password"].(string),
		Host:     data["host"].(string),
		Port:     data["port"].(string),
		Name:     data["name"].(string),
	}, nil
}

func (c *DBConfig) ConnString() string {
	return fmt.Sprintf(
		"postgres://%s:%s@%s:%s/%s?sslmode=disable",
		c.User, c.Password, c.Host, c.Port, c.Name,
	)
}

package db

import (
	"context"
	"fmt"

	"github.com/jackc/pgx/v5/pgxpool"
)

func Init(url string) (*pgxpool.Pool, error) {
	pool, err := pgxpool.New(context.Background(), url)
	if err != nil {
		return nil, err
	}
	return pool, nil
}

func Migrate(db *pgxpool.Pool) error {
	query := `
	CREATE TABLE IF NOT EXISTS users (
		id SERIAL PRIMARY KEY,
		name TEXT NOT NULL,
		email TEXT UNIQUE NOT NULL,
		created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
	);
	`

	_, err := db.Exec(context.Background(), query)
	if err != nil {
		return fmt.Errorf("migration failed: %w", err)
	}

	return nil
}

package handler

import (
	"encoding/json"
	"net/http"
	"service1/internal/metrics"
	"service1/internal/middlewares"
	"service1/internal/models"
	"service1/internal/service"
	"strconv"

	"github.com/go-chi/chi/middleware"
	"github.com/go-chi/chi/v5"
	"github.com/prometheus/client_golang/prometheus/promhttp"
	"go.opentelemetry.io/otel/trace"
)

type UserHandler struct {
	svc service.UserService
	tr  trace.Tracer
}

func NewUserHandler(s service.UserService, tr trace.Tracer) *UserHandler {
	return &UserHandler{svc: s, tr: tr}
}

func (h *UserHandler) Routes(r chi.Router) {
	metrics.Init()

	r.Use(middleware.Logger)
	r.Use(middleware.Recoverer)
	r.Use(middlewares.PrometheusMiddleware)

	r.Handle("/metrics", promhttp.Handler())
	r.Route("/users", func(r chi.Router) {

		r.Post("/", h.create)
		r.Get("/", h.list)
		r.Get("/{id}", h.get)
		r.Put("/{id}", h.update)
		r.Delete("/{id}", h.delete)
	})
}

func (h *UserHandler) create(w http.ResponseWriter, r *http.Request) {
	ctx, span := h.tr.Start(r.Context(), "CreateUser")
	defer span.End()

	var u models.User

	err := json.NewDecoder(r.Body).Decode(&u)
	if err != nil {
		http.Error(w, "bad json: "+err.Error(), 400)
		return
	}

	if err := h.svc.Create(ctx, &u); err != nil {
		http.Error(w, err.Error(), 400)
		return
	}

	json.NewEncoder(w).Encode(u)
}

func (h *UserHandler) list(w http.ResponseWriter, r *http.Request) {
	data, _ := h.svc.List(r.Context())
	json.NewEncoder(w).Encode(data)
}

func (h *UserHandler) get(w http.ResponseWriter, r *http.Request) {
	id, _ := strconv.Atoi(chi.URLParam(r, "id"))
	u, err := h.svc.Get(r.Context(), id)
	if err != nil {
		http.Error(w, "not found", 404)
		return
	}
	json.NewEncoder(w).Encode(u)
}

func (h *UserHandler) update(w http.ResponseWriter, r *http.Request) {
	id, _ := strconv.Atoi(chi.URLParam(r, "id"))

	var u models.User
	json.NewDecoder(r.Body).Decode(&u)
	u.ID = id

	h.svc.Update(r.Context(), &u)
	w.WriteHeader(http.StatusOK)
}

func (h *UserHandler) delete(w http.ResponseWriter, r *http.Request) {
	id, _ := strconv.Atoi(chi.URLParam(r, "id"))
	h.svc.Delete(r.Context(), id)
	w.WriteHeader(http.StatusOK)
}

package kafka

import (
	"context"
	"encoding/json"

	"github.com/segmentio/kafka-go"
)

type Producer struct {
	writer *kafka.Writer
}

func NewProducer(broker string, topic string) *Producer {
	return &Producer{
		writer: &kafka.Writer{
			Addr:     kafka.TCP(broker),
			Topic:    topic,
			Balancer: &kafka.LeastBytes{},
		},
	}
}

func (p *Producer) Send(ctx context.Context, key string, value any) error {
	data, err := json.Marshal(value)
	if err != nil {
		return err
	}

	return p.writer.WriteMessages(ctx, kafka.Message{
		Key:   []byte(key),
		Value: data,
	})
}

package middlewares

import (
	"net/http"
	"service1/internal/metrics"
	"time"

	"github.com/go-chi/chi"
)

type statusRecorder struct {
	http.ResponseWriter
	status int
}

func (r *statusRecorder) WriteHeader(code int) {
	r.status = code
	r.ResponseWriter.WriteHeader(code)
}

func PrometheusMiddleware(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		rec := &statusRecorder{
			ResponseWriter: w,
			status:         http.StatusOK,
		}

		if r.URL.Path == "/metrics" {
			next.ServeHTTP(w, r)
			return
		}
		start := time.Now()

		next.ServeHTTP(rec, r) // ✅ only once

		duration := time.Since(start).Seconds()

		rctx := chi.RouteContext(r.Context())

		route := "unknown"
		if rctx != nil {
			if pattern := rctx.RoutePattern(); pattern != "" {
				route = pattern
			}
		}

		metrics.HttpDuration.WithLabelValues(
			r.Method,
			route,
		).Observe(duration)

		metrics.HttpRequests.WithLabelValues(
			r.Method,
			route,
			http.StatusText(rec.status),
		).Inc()
	})
}

package otelsetup

import (
	"context"
	"log"

	"go.opentelemetry.io/otel"
	"go.opentelemetry.io/otel/exporters/jaeger"
	"go.opentelemetry.io/otel/sdk/resource"
	sdktrace "go.opentelemetry.io/otel/sdk/trace"
)

func InitTracer() func() {
	exp, err := jaeger.New(jaeger.WithCollectorEndpoint(
		jaeger.WithEndpoint("http://localhost:14268/api/traces"),
	))
	if err != nil {
		log.Fatal(err)
	}

	tp := sdktrace.NewTracerProvider(
		sdktrace.WithBatcher(exp),
		sdktrace.WithResource(resource.Default()),
	)

	otel.SetTracerProvider(tp)

	return func() {
		_ = tp.Shutdown(context.Background())
	}
}

package repository

import (
	"context"
	"service1/internal/models"

	"github.com/jackc/pgx/v5/pgxpool"
)

type UserRepository interface {
	Create(ctx context.Context, u *models.User) error
	GetAll(ctx context.Context) ([]models.User, error)
	GetByID(ctx context.Context, id int) (*models.User, error)
	Update(ctx context.Context, u *models.User) error
	Delete(ctx context.Context, id int) error
}

type userRepo struct {
	db *pgxpool.Pool
}

func NewUserRepo(db *pgxpool.Pool) UserRepository {
	return &userRepo{db: db}
}

func (r *userRepo) Create(ctx context.Context, u *models.User) error {
	return r.db.QueryRow(ctx,
		"INSERT INTO users(name,email) VALUES($1,$2) RETURNING id",
		u.Name, u.Email).Scan(&u.ID)
}

func (r *userRepo) GetAll(ctx context.Context) ([]models.User, error) {
	rows, err := r.db.Query(ctx, "SELECT id,name,email FROM users")
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var list []models.User
	for rows.Next() {
		var u models.User
		rows.Scan(&u.ID, &u.Name, &u.Email)
		list = append(list, u)
	}
	return list, nil
}
func (r *userRepo) GetByID(ctx context.Context, id int) (*models.User, error) {
	var u models.User
	err := r.db.QueryRow(ctx,
		"SELECT id,name,email FROM users WHERE id=$1", id).
		Scan(&u.ID, &u.Name, &u.Email)
	return &u, err
}

func (r *userRepo) Update(ctx context.Context, u *models.User) error {
	_, err := r.db.Exec(ctx,
		"UPDATE users SET name=$1,email=$2 WHERE id=$3",
		u.Name, u.Email, u.ID)
	return err
}

func (r *userRepo) Delete(ctx context.Context, id int) error {
	_, err := r.db.Exec(ctx, "DELETE FROM users WHERE id=$1", id)
	return err
}

package service

import (
	"context"
	"errors"
	"log/slog"
	"service1/internal/events"
	"service1/internal/kafka"
	"service1/internal/logger"
	"service1/internal/models"
	"service1/internal/repository"
)

type UserService interface {
	Create(ctx context.Context, u *models.User) error
	List(ctx context.Context) ([]models.User, error)
	Get(ctx context.Context, id int) (*models.User, error)
	Update(ctx context.Context, u *models.User) error
	Delete(ctx context.Context, id int) error
}

type userService struct {
	repo     repository.UserRepository
	producer *kafka.Producer
}

var ErrInvalidInput = errors.New("invalid input")

func NewUserService(r repository.UserRepository, p *kafka.Producer) UserService {
	return &userService{repo: r, producer: p}
}

func (s *userService) Create(ctx context.Context, u *models.User) error {
	// example validation
	if u.Email == "" {
		return ErrInvalidInput
	}

	if err := s.repo.Create(ctx, u); err != nil {
		return err
	}

	// emit event
	event := events.UserCreatedEvent{
		ID:    u.ID,
		Name:  u.Name,
		Email: u.Email,
	}
	// fire-and-forget (basic)
	if err := s.producer.Send(ctx, "user_created", event); err != nil {
		logger.Log.Error("migration failed", slog.Any("error", err))

	}

	return nil
}

func (s *userService) List(ctx context.Context) ([]models.User, error) {
	return s.repo.GetAll(ctx)
}

func (s *userService) Get(ctx context.Context, id int) (*models.User, error) {
	return s.repo.GetByID(ctx, id)
}

func (s *userService) Update(ctx context.Context, u *models.User) error {
	return s.repo.Update(ctx, u)
}

func (s *userService) Delete(ctx context.Context, id int) error {
	return s.repo.Delete(ctx, id)
}
