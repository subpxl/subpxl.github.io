File: main.go

package main

import (
	"calculator-app/Usecase/calculation"
	"calculator-app/Usecase/calculationwithanalytics"
	"calculator-app/config"
	"calculator-app/domain/calculator"
	"calculator-app/infrastructure/database"
	analyticsRepo "calculator-app/infrastructure/repository/analytics"
	"log"

	"gorm.io/gorm"
)

type DBType string

const (
	PostgreSQL DBType = "postgres"
	SQLite     DBType = "sqlite"
)

func initDB(config *config.Config) (*gorm.DB, error) {
	switch config.Database.Type {
	case "postgres":
		return database.NewPostgresConnection(
			config.Database.Host,
			config.Database.Port,
			config.Database.User,
			config.Database.Password,
			config.Database.DBName,
		)
	case "sqlite":
		return database.NewSQLiteConnection(config.Database.SQLitePath)
	default:
		return database.NewSQLiteConnection(config.Database.SQLitePath)
	}
}

func main() {
	cfg := config.NewConfig()

	// Initialize database connection
	db, err := initDB(cfg)
	if err != nil {
		log.Fatal(err)
	}

	// initiate repositories
	analyticsRepository, err := analyticsRepo.NewGormRepository(db)
	if err != nil {
		log.Fatal(err)
	}

	// initiate calculator
	calc := calculator.NewCalculator() // This returns *Calculator

	// initialize usecases
	calculateWithAnalytics := calculationwithanalytics.New(analyticsRepository, calc)
	calculateUseCase := calculation.New(calc)

	// use plain calculator
	result1, err := calculateUseCase.Add(1, 2)
	if err != nil {
		log.Fatal(err)
	}
	log.Printf("Plain calculation result: %f\n", result1)
	result, err := calculateWithAnalytics.AddWithAnalytics(1, 2)
	if err != nil {
		log.Fatal(err)
	}
	log.Printf("Result: %f\n", result)
}

File: calculator.go

package calculator

import "errors"

var (
	ErrDivideByZero = errors.New("cannot divide by zero")
)

type Input struct {
	X float64
	Y float64
}

type Result struct {
	Value float64
}

type Calculator struct {
}

func NewCalculator() *Calculator {
    return &Calculator{}
}


func (c *Calculator) Add(input Input) (Result, error) {
	return Result{input.X + input.Y}, nil
}

func (c *Calculator) Subtract(input Input) (Result, error) {
	return Result{input.X - input.Y}, nil

}

func (c *Calculator) Multiply(input Input) (Result, error) {
	return Result{input.X * input.Y}, nil
}

func (c *Calculator) Divide(input Input) (Result, error) {
	if input.Y == 0 {
		return Result{}, ErrDivideByZero
	}
	return Result{input.X / input.Y}, nil
}

func (c *Calculator) Remainder(input Input) (Result, error) {
	if input.Y == 0 {
		return Result{}, ErrDivideByZero
	}
	return Result{Value: float64(int(input.X) % int(input.Y))}, nil
}

func (c *Calculator) Quotient(input Input) (Result, error) {
	if input.Y == 0 {
		return Result{}, ErrDivideByZero
	}
	return Result{Value: float64(int(input.X) / int(input.Y))}, nil
}

package database

import (
	"fmt"

	"gorm.io/driver/postgres"

	"gorm.io/gorm"
)

func NewPostgresConnection(host string,
	port int,
	user string,
	password string,
	dbname string) (*gorm.DB, error) {
	dsn := fmt.Sprintf(
		"host=%s user=%s password=%s dbname=%s port=%d sslmode=disable",
		host, user, password, dbname, port,
	)

	return gorm.Open(postgres.Open(dsn), &gorm.Config{})
}

File: gorm_repository.go

package analytics

import (
	"calculator-app/domain/analytics"

	"gorm.io/gorm"
)

type GormRepository struct {
	db *gorm.DB
}

func NewGormRepository(db *gorm.DB) (*GormRepository, error) {
	err := db.AutoMigrate(&analytics.CalculationRecord{})
	if err != nil {
		return nil, err
	}
	return &GormRepository{db}, nil
}

func (r *GormRepository) CreateRecord(calculationRecord analytics.CalculationRecord) error {
	return r.db.Create(&calculationRecord).Error
}

func (r *GormRepository) ListRecord() ([]analytics.CalculationRecord, error) {
	var records []analytics.CalculationRecord
	err := r.db.Find(&records).Error
	return records, err
}

// usecase/calculationwithanalytics/calculationwithanalytics.go
package calculationwithanalytics

import (
	"calculator-app/usecase" // Add this import

	"calculator-app/domain/analytics"
	"calculator-app/domain/calculator"
	"encoding/json"
	"time"

	"github.com/google/uuid"
)

type calculationwithanalytics struct {
	repo analytics.Repository
	calc *calculator.Calculator
}

func New(repo analytics.Repository, calc *calculator.Calculator) usecase.CalculatorWithAnalyticsUseCase {
	return &calculationwithanalytics{
		repo: repo,
		calc: calc,
	}
}

func (c *calculationwithanalytics) AddWithAnalytics(x, y float64) (float64, error) {
	input := calculator.Input{
		X: x,
		Y: y,
	}
	result, err := c.calc.Add(input)
	if err != nil {
		return 0, err
	}

	inputJSON, _ := json.Marshal(input)
	outputJSON, _ := json.Marshal(result)

	record := analytics.CalculationRecord{
		ID:        uuid.New().String(),
		Operation: "ADD",
		Input:     string(inputJSON),
		Output:    string(outputJSON),
		TimeStamp: time.Now(),
	}

	if err := c.repo.CreateRecord(record); err != nil {
		return 0, err
	}
	return result.Value, nil
}
