package main

import (
    "inventory/controllers"
    "inventory/models"

    "github.com/gin-gonic/gin"
    "gorm.io/driver/sqlite"
    "gorm.io/gorm"
)

func main() {
    db, err := gorm.Open(sqlite.Open("inventory.db"), &gorm.Config{})
    if err != nil {
        panic(err)
    }

    db.AutoMigrate(&models.Product{})

    r := gin.Default()

    pc := controllers.NewProductController(db)

    r.GET("/", pc.Index)
    r.GET("/products/new", pc.New)
    r.POST("/products", pc.Create)
    r.GET("/products/:id/edit", pc.Edit)
    r.POST("/products/:id", pc.Update)
    r.POST("/products/:id/delete", pc.Delete)

    r.Run(":8080")
}

File: product_controller.go

package controllers

import (
	"inventory/models"
	"inventory/views"
	"net/http"
	"strconv"

	"github.com/gin-gonic/gin"
	"gorm.io/gorm"
)

type ProductController struct {
	db *gorm.DB
}

func NewProductController(db *gorm.DB) *ProductController {
	return &ProductController{db: db}
}

func (pc *ProductController) Index(c *gin.Context) {
	var products []models.Product
	pc.db.Find(&products)
	views.ProductsIndex(products).Render(c.Request.Context(), c.Writer)
}

func (pc *ProductController) New(c *gin.Context) {
	views.ProductsForm(models.Product{}, "/products").Render(c.Request.Context(), c.Writer)
}

func (pc *ProductController) Create(c *gin.Context) {
	qty, _ := strconv.Atoi(c.PostForm("quantity"))
	price, _ := strconv.ParseFloat(c.PostForm("price"), 64)

	p := models.Product{
		Name:     c.PostForm("name"),
		Quantity: qty,
		Price:    price,
	}

	pc.db.Create(&p)
	c.Redirect(http.StatusSeeOther, "/")
}

func (pc *ProductController) Edit(c *gin.Context) {
	var product models.Product
	pc.db.First(&product, c.Param("id"))
	views.ProductsForm(product, "/products/"+c.Param("id")).Render(c.Request.Context(), c.Writer)
}

func (pc *ProductController) Update(c *gin.Context) {
	var product models.Product
	pc.db.First(&product, c.Param("id"))

	product.Name = c.PostForm("name")
	product.Quantity, _ = strconv.Atoi(c.PostForm("quantity"))
	product.Price, _ = strconv.ParseFloat(c.PostForm("price"), 64)

	pc.db.Save(&product)
	c.Redirect(http.StatusSeeOther, "/")
}

func (pc *ProductController) Delete(c *gin.Context) {
	pc.db.Delete(&models.Product{}, c.Param("id"))
	c.Redirect(http.StatusSeeOther, "/")
}
