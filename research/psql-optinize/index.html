package main

import (
	"context"
	"fmt"
	"math/rand"
	"time"

	"github.com/jackc/pgx/v5"
	"github.com/jackc/pgx/v5/pgxpool"
)

// ── Configuration ───────────────────────────────────────────
const (
	numCustomers     = 1000
	numProducts      = 5000
	numOrders        = 300000
	numReviews       = 200000
	numInventoryLogs = 100000

	// Update your connection string here
	postgresDSN = "postgres://user:password@localhost:5432/testdb?sslmode=disable"
)

var (
	countries = []string{"US", "IN", "UK", "CA", "DE", "JP", "FR", "AU", "BR", "SG"}
	segments  = []string{"free", "basic", "premium", "enterprise"}
	statuses  = []string{"pending", "paid", "shipped", "cancelled"}
	reasons   = []string{"sale", "restock", "adjustment"}
	categories = []string{"Electronics", "Clothing", "Books", "Toys", "Sports", "Beauty", "Automotive"}
)

func main() {
	ctx := context.Background()
	rand.Seed(time.Now().UnixNano())

	// Initialize Connection Pool
	config, err := pgxpool.ParseConfig(postgresDSN)
	must(err)
	db, err := pgxpool.NewWithConfig(ctx, config)
	must(err)
	defer db.Close()

	fmt.Println("=== Seeding start (Using pgx/v5) ===")

	catIDs := seedCategories(ctx, db)
	prodIDs := seedProducts(ctx, db, catIDs)
	custIDs := seedCustomers(ctx, db)
	
	// Large datasets
	orderIDs := seedOrders(ctx, db, custIDs)
	seedOrderItems(ctx, db, orderIDs, prodIDs)
	seedReviews(ctx, db, custIDs, prodIDs)
	seedInventoryLog(ctx, db, prodIDs)

	fmt.Println("=== Seeding complete ===")
}

// ── Categories (Simple Inserts) ─────────────────────────────
func seedCategories(ctx context.Context, db *pgxpool.Pool) []int {
	var ids []int
	for _, name := range categories {
		var id int
		err := db.QueryRow(ctx, "INSERT INTO categories (name) VALUES ($1) RETURNING id", name).Scan(&id)
		must(err)
		ids = append(ids, id)
	}
	return ids
}

// ── Products (Inserts with Returning) ───────────────────────
func seedProducts(ctx context.Context, db *pgxpool.Pool, catIDs []int) []int {
	var ids []int
	base := time.Now().AddDate(-2, 0, 0)
	for i := 1; i <= numProducts; i++ {
		var id int
		err := db.QueryRow(ctx, 
			"INSERT INTO products (name, category_id, price, stock, created_at) VALUES ($1,$2,$3,$4,$5) RETURNING id",
			fmt.Sprintf("Product_%d", i),
			catIDs[rand.Intn(len(catIDs))],
			float64(rand.Intn(49000)+1000)/100,
			rand.Intn(500),
			base.Add(randDuration(2*365*24)),
		).Scan(&id)
		must(err)
		ids = append(ids, id)
		logProgress("products", i, numProducts)
	}
	return ids
}

// ── Customers (Inserts with Returning) ──────────────────────
func seedCustomers(ctx context.Context, db *pgxpool.Pool) []int {
	var ids []int
	base := time.Now().AddDate(-3, 0, 0)
	for i := 1; i <= numCustomers; i++ {
		var id int
		err := db.QueryRow(ctx, 
			"INSERT INTO customers (name, email, country, segment, created_at) VALUES ($1,$2,$3,$4,$5) RETURNING id",
			fmt.Sprintf("Customer_%d", i),
			fmt.Sprintf("cust_%d@example.com", i),
			countries[rand.Intn(len(countries))],
			segments[rand.Intn(len(segments))],
			base.Add(randDuration(3*365*24)),
		).Scan(&id)
		must(err)
		ids = append(ids, id)
		logProgress("customers", i, numCustomers)
	}
	return ids
}

// ── Orders (COPY with Batching) ──────────────────────────────
func seedOrders(ctx context.Context, db *pgxpool.Pool, custIDs []int) []int {
	base := time.Now().AddDate(-1, 0, 0)
	var rows [][]interface{}
	
	for i := 0; i < numOrders; i++ {
		rows = append(rows, []interface{}{
			custIDs[rand.Intn(len(custIDs))],
			statuses[rand.Intn(len(statuses))],
			base.Add(randDuration(365 * 24)),
		})
		
		// Flush every 50k rows to keep memory low and DB happy
		if len(rows) >= 50000 {
			_, err := db.CopyFrom(ctx, pgx.Identifier{"orders"}, []string{"customer_id", "status", "created_at"}, pgx.CopyFromRows(rows))
			must(err)
			rows = nil
			logProgress("orders copy", i+1, numOrders)
		}
	}
	if len(rows) > 0 {
		_, err := db.CopyFrom(ctx, pgx.Identifier{"orders"}, []string{"customer_id", "status", "created_at"}, pgx.CopyFromRows(rows))
		must(err)
	}

	// Fetch IDs without ORDER BY for speed
	res, err := db.Query(ctx, "SELECT id FROM orders")
	must(err)
	defer res.Close()
	var ids []int
	for res.Next() {
		var id int
		must(res.Scan(&id))
		ids = append(ids, id)
	}
	return ids
}

// ── Order Items (COPY with Batching) ────────────────────────
func seedOrderItems(ctx context.Context, db *pgxpool.Pool, orderIDs, prodIDs []int) {
	prices := fetchPrices(ctx, db)
	var rows [][]interface{}
	totalCount := 0

	for _, oID := range orderIDs {
		itemsInOrder := rand.Intn(3) + 1
		for j := 0; j < itemsInOrder; j++ {
			pID := prodIDs[rand.Intn(len(prodIDs))]
			rows = append(rows, []interface{}{oID, pID, rand.Intn(5) + 1, prices[pID]})
			totalCount++
		}

		if len(rows) >= 50000 {
			_, err := db.CopyFrom(ctx, pgx.Identifier{"order_items"}, []string{"order_id", "product_id", "quantity", "unit_price"}, pgx.CopyFromRows(rows))
			must(err)
			rows = nil
			logProgress("order_items copy", totalCount, -1)
		}
	}
	if len(rows) > 0 {
		db.CopyFrom(ctx, pgx.Identifier{"order_items"}, []string{"order_id", "product_id", "quantity", "unit_price"}, pgx.CopyFromRows(rows))
	}
}

// ── Reviews & Inventory (Simplified COPY) ────────────────────
func seedReviews(ctx context.Context, db *pgxpool.Pool, custIDs, prodIDs []int) {
	seen := make(map[[2]int]bool)
	var rows [][]interface{}
	base := time.Now().AddDate(-1, 0, 0)

	for len(rows) < numReviews {
		cID := custIDs[rand.Intn(len(custIDs))]
		pID := prodIDs[rand.Intn(len(prodIDs))]
		if seen[[2]int{cID, pID}] { continue }
		seen[[2]int{cID, pID}] = true

		rows = append(rows, []interface{}{cID, pID, rand.Intn(5)+1, base.Add(randDuration(365*24))})
		
		if len(rows) % 50000 == 0 {
			_, err := db.CopyFrom(ctx, pgx.Identifier{"reviews"}, []string{"customer_id", "product_id", "rating", "created_at"}, pgx.CopyFromRows(rows[len(rows)-50000:]))
			must(err)
			logProgress("reviews copy", len(rows), numReviews)
		}
	}
}

func seedInventoryLog(ctx context.Context, db *pgxpool.Pool, prodIDs []int) {
	var rows [][]interface{}
	base := time.Now().AddDate(-2, 0, 0)
	for i := 0; i < numInventoryLogs; i++ {
		rows = append(rows, []interface{}{
			prodIDs[rand.Intn(len(prodIDs))],
			rand.Intn(100) - 20,
			reasons[rand.Intn(len(reasons))],
			base.Add(randDuration(2*365*24)),
		})
	}
	_, err := db.CopyFrom(ctx, pgx.Identifier{"inventory_log"}, []string{"product_id", "delta", "reason", "created_at"}, pgx.CopyFromRows(rows))
	must(err)
}

// ── Helpers ──────────────────────────────────────────────────
func fetchPrices(ctx context.Context, db *pgxpool.Pool) map[int]float64 {
	rows, _ := db.Query(ctx, "SELECT id, price FROM products")
	defer rows.Close()
	m := make(map[int]float64)
	for rows.Next() {
		var id int
		var p float64
		rows.Scan(&id, &p)
		m[id] = p
	}
	return m
}

func randDuration(hours int) time.Duration {
	return time.Duration(rand.Intn(hours*3600)) * time.Second
}

func logProgress(label string, current, total int) {
	if total == -1 {
		fmt.Printf("  [%s] Processed: %d\n", label, current)
		return
	}
	if current%(total/10) == 0 || current == total {
		fmt.Printf("  [%s] %d / %d\n", label, current, total)
	}
}

func must(err error) {
	if err != nil {
		panic(err)
	}
}
